'use strict';
var _ = require('lodash');
var sprintf = require('sprintf-js').sprintf;
 
var get_type = require('./get_type');
 
module.exports = function(site, templates, content, callback){
    get_type(site, content, function(err, type){
        var template = type + site.default_template_extension;
        err = null;
        if (_.has(content.meta, 'template') && _.isString(content.meta.template) && 0 < content.meta.template.trim().length){
            template = content.meta.template.trim();
        }
 
        if (!_.has(templates, template)){
            err = new Error(sprintf('The template for "%s" for %s is missing.', template, content.relative_path));
            callback(err, null);
        } else {
            callback(err, templates[template]);
        }
    })
};
